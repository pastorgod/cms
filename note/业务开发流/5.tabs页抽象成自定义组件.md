# tabs页抽象成自定义组件

```vue
<template>
  <div class="TabSlot">
    <span class="tabs-icon">
      <el-icon :size="size">
        <component :is="resolveComponent(icon)" />
      </el-icon>
    </span>
    <span>{{ label }}</span>
  </div>
</template>

<!-- =============================代码============================ -->
<script setup lang="ts">
import { ref, computed, watch, onMounted, resolveComponent } from 'vue'

// 参数有默认值的话，可以使用 withDefaults 包裹下，第一个参数为 props，第二个参数为默认值
const props = withDefaults(defineProps<{
  icon: string
  label: string
  size: number
}>(), {
  size: 16  // size 默认值为 16
})

</script>
```
