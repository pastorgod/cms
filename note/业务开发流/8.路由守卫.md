# 路由守卫

```ts
//导航守卫
// import loginStore from '@/stores/loginStore'
router.beforeEach((to, from, next) => {
    if (to.name === 'login') {
        next()
        return
    }

    //内存中没有登录信息
    if (IsNullOrEmpty(loginStore.username) || IsNullOrEmpty(loginStore.projectId)) {
        next({ name: 'login' })
        return
    }

    next()
})

```
