# 可组合函数

想找下全局函数的定义和使用，但是没找到合适的，不是全局属性和全局方法，就是依赖注入，
使用起来不是很方便
vue更推荐组合函数，用法如下

```ts
export function stringUtils() {
    
    //字符串判空
    const IsEmpty = (str: string | undefined): boolean => {
      return (!str || str.trim() === "");
    }
    
    // 比较两个值是否相等
    const isEqual = (a: any, b: any): boolean => {
    return JSON.stringify(a) === JSON.stringify(b)
  }
  return {
    IsEmpty,
    isEqual
  }
}
```
